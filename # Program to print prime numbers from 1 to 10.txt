# Program to print prime numbers from 1 to 10

# Loop through numbers from 1 to 10
for num in range(1, 11):

    # Prime numbers are greater than 1
    if num > 1:

        # Assume the current number is prime
        is_prime = True
        
        # Check divisibility from 2 to num-1
        for i in range(2, num):

            # If num is divisible by i, it is not prime
            if num % i == 0:
                is_prime = False
                break   # Stop checking further divisors

        # If the number is still marked as prime, print it
        if is_prime:
            print(num)
# sample output
